NodeJS Scoring Logic
class ProfessionalCoachabilityAssessment {
  constructor(questionBank, responses) {
    this.questionBank = questionBank;
    this.responses = responses;
  }


  evaluate() {
    let total = 0;
    let max = 0;
    let dimensions = {};


    this.questionBank.forEach(q => {
      const selected = q.options.find(
        o => o.key === this.responses[q.id]
      );


      total += selected.weight;
      max += 4;


      if (!dimensions[q.dimension]) {
        dimensions[q.dimension] = {
          score: 0,
          max: 0
        };
      }


      dimensions[q.dimension].score += selected.weight;
      dimensions[q.dimension].max += 4;
    });


    const overall = Math.round((total / max) * 100);


    const profile = {};
    Object.keys(dimensions).forEach(dim => {
      profile[dim] = Math.round(
        (dimensions[dim].score / dimensions[dim].max) * 100
      );
    });


    return {
      score: overall,
      percentile: this.percentile(overall),
      coachabilityBand: this.band(overall),
      learningReadiness: this.readiness(overall),
      profile,
      narrative: this.narrative(overall, profile)
    };
  }


  percentile(score){
    if(score>=90) return 95;
    if(score>=80) return 85;
    if(score>=70) return 70;
    if(score>=60) return 55;
    return 35;
  }


  band(score){
    if(score>=90) return "Highly Coachable Growth Leader";
    if(score>=80) return "Strong Learner";
    if(score>=70) return "Trainable Professional";
    if(score>=60) return "Moderately Receptive";
    return "Resistant / Fixed Pattern";
  }


  readiness(score){
    if(score>=85) return "Ready for accelerated development";
    if(score>=70) return "Ready with coaching";
    return "Needs mindset intervention";
  }


  narrative(score, profile){
    return {
      summary:
        `Professional demonstrates ${this.band(score)} orientation toward coaching, learning agility, and career adaptability.`,
      strongestDimension:
        Object.keys(profile).reduce((a,b)=>profile[a]>profile[b]?a:b),
      weakestDimension:
        Object.keys(profile).reduce((a,b)=>profile[a]<profile[b]?a:b)
    };
  }
}


module.exports = ProfessionalCoachabilityAssessment;
